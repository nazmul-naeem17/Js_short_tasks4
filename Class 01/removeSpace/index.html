Make a function that removes any space from the string

function spaceRemove(str) { let result = ''; for (let i = 0; i < str.length; i++) { if (str[i] !== ' ') { result += str[i]; } } return result; } let stringWithSpaces = 'My name is Nazmul Hasan'; stringWithoutSpaces = spaceRemove(stringWithSpaces); console.log(stringWithoutSpaces);//Output: MynameisNazmulHasan
