Make a function that counts the number of vowels in a string without ‘a’

function countVowels(str) { let count = 0; let vowels = ['e', 'i', 'o', 'u']; for (let i = 0; i < str.length; i++) { if (vowels.includes(str[i])) { count++; } } return count; } let inputString = 'Hello, world!'; let vowelCount = countVowels(inputString); console.log(vowelCount);//Output:3
